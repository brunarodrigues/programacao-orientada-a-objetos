salario = float(input("Quanto você ganha? "))
if salario <= 280:
  reajuste = (salario * 0.20) 
  novo = reajuste + salario
  print("Salário antes do reajuste: {:.2f}".format(salario))
  print("Percentual de aumento aplicado foi de: 20%")
  print("Valor do aumento foi de: {:.2f}".format(reajuste))
  print("o novo salário, após o aumento vai ficar: {:.2f}".format(novo))
elif salario > 280 and salario <= 700:
  reajuste = (salario * 0.15) 
  novo = reajuste + salario
  print("Salário antes do reajuste:  {:.2f}".format(salario))
  print("Percentual de aumento aplicado foi de: 15%")
  print("Valor do aumento foi de: {:.2f}".format(reajuste))
  print("Novo salário, após o aumento vai ficar: {:.2f}".format(novo))
elif salario > 700 and salario <= 1500:
  reajuste = (salario * 0.10) 
  novo = reajuste + salario
  print("Salário antes do reajuste : {:.2f}".format(salario))
  print("Percentual de aumento aplicado foi de: 10%")
  print("Valor do aumento foi de: {:.2f}".format(reajuste))
  print("Novo salário, após o aumento vai ficar: {:.2f}".format(novo))
else:

  novo = reajuste + salario

  print("Novo salário, após o aumento vai ficar: {:.2f}".format(novo))
